   # Build Instructions

   ## Prerequisites

   1. **Windows 10/11** with Windows SDK installed
   2. **Rust toolchain** with MSVC target:
      ```bash
      rustup default stable-x86_64-pc-windows-msvc
      ```

   3. **Node.js 18+** for frontend tooling

   4. **OpenCV 4.8+** (REQUIRED for build)

   ## OpenCV Setup

   ### Option 1: vcpkg (Recommended)

   ```bash
   # Install vcpkg
   git clone https://github.com/Microsoft/vcpkg.git C:\vcpkg
   cd C:\vcpkg
   .\bootstrap-vcpkg.bat
   .\vcpkg integrate install

   # Install OpenCV (64-bit, static libraries)
   .\vcpkg install opencv:x64-windows-static
   ```

   Set environment variables (permanent in System Properties):
   ```
   OPENCV_LINK_PATHS=C:\vcpkg\installed\x64-windows-static\lib
   OPENCV_INCLUDE_PATHS=C:\vcpkg\installed\x64-windows-static\include
   ```

   ### Option 2: Manual Installation

   1. Download OpenCV 4.8.0 Windows prebuilt binaries
   2. Extract to `C:\opencv`
   3. Set environment variables:
      ```
      OPENCV_LINK_PATHS=C:\opencv\build\x64\vc16\lib
      OPENCV_INCLUDE_PATHS=C:\opencv\build\include
      ```

   ### Verify Setup

   ```bash
   cargo build --manifest-path src-tauri/Cargo.toml
   ```

   If you see linker errors about OpenCV, check:
   - Environment variables are set correctly
   - OpenCV architecture matches Rust target (x64)
   - OpenCV version is 4.8+

   ## Building

   ```bash
   # Install frontend dependencies
   npm install

   # Development (hot reload)
   cargo tauri dev

   # Release build
   cargo tauri build
   ```

   ## Troubleshooting

   **Build fails with "cannot find -lopencv_core"**: OpenCV environment variables not set correctly.

   **Build fails with "unresolved external symbol"**: OpenCV version mismatch or architecture mismatch.

   See [PITFALLS.md](.planning/research/PITFALLS.md) for known issues.

   use serde::{Serialize, Deserialize};

   /// Internal coordinate: Physical (device) pixels
   /// This is what capture APIs return and what we store in config
   #[derive(Serialize, Deserialize, Clone, Copy, Debug, PartialEq, Eq, Hash)]
   pub struct PhysicalCoord {
       pub x: i32,
       pub y: i32,
   }

   /// Display coordinate: User-visible (DPI-scaled) pixels
   /// This is what UI displays and what users interact with
   #[derive(Serialize, Deserialize, Clone, Copy, Debug, PartialEq, Eq)]
   pub struct DisplayCoord {
       pub x: i32,
       pub y: i32,
   }

   /// DPI information for a display
   #[derive(Serialize, Deserialize, Clone, Debug)]
   pub struct DpiInfo {
       pub display_id: String,
       pub scale_factor: f64, // 1.0 = 100%, 1.5 = 150%, 2.0 = 200%
   }

   /// Convert display coordinates to physical coordinates
   pub fn to_physical(display: DisplayCoord, scale: f64) -> PhysicalCoord {
       PhysicalCoord {
           x: (display.x as f64 * scale).round() as i32,
           y: (display.y as f64 * scale).round() as i32,
       }
   }

   /// Convert physical coordinates to display coordinates
   pub fn to_display(physical: PhysicalCoord, scale: f64) -> DisplayCoord {
       DisplayCoord {
           x: (physical.x as f64 / scale).round() as i32,
           y: (physical.y as f64 / scale).round() as i32,
       }
   }

   /// Check if ROI coordinates need DPI invalidation
   pub fn check_dpi_invalidation(
       roi_physical_rect: &crate::models::Rect,
       current_dpi: &DpiInfo,
       stored_flags: &crate::models::DpiInvalidationFlags,
   ) -> DpiInvalidationResult {
       // D-11: Check if DPI scale changed
       let scale_changed = (stored_flags.last_dpi_scale - current_dpi.scale_factor).abs() > 0.01;

       // D-11: Check if display changed
       let display_changed = stored_flags.last_display_id.as_ref()
           .map(|id| id != &current_dpi.display_id)
           .unwrap_or(true);

       let invalid = scale_changed || display_changed;

       DpiInvalidationResult {
           invalid,
           reason: if invalid {
               if scale_changed { Some("DPI 缩放已更改".to_string()) }
               else { Some("显示器已更改".to_string()) }
           } else {
               None
           },
       }
   }

   #[derive(Serialize, Deserialize, Clone, Debug)]
   pub struct DpiInvalidationResult {
       pub invalid: bool,
       pub reason: Option<String>, // Chinese reason message
   }

   /// Get DPI information for the current display
   pub fn get_current_dpi() -> Option<DpiInfo> {
       // Implementation will use Windows API in Phase 2
       // For now, return a default
       Some(DpiInfo {
           display_id: "default".to_string(),
           scale_factor: 1.0, // Default to 100%
       })
   }

   #[cfg(test)]
   mod tests {
       use super::*;

       #[test]
       fn test_to_physical() {
           let display = DisplayCoord { x: 100, y: 100 };
           let physical = to_physical(display, 1.5); // 150% DPI
           assert_eq!(physical.x, 150);
           assert_eq!(physical.y, 150);
       }

       #[test]
       fn test_to_display() {
           let physical = PhysicalCoord { x: 150, y: 150 };
           let display = to_display(physical, 1.5);
           assert_eq!(display.x, 100);
           assert_eq!(display.y, 100);
       }

       #[test]
       fn test_dpi_invalidation_detection() {
           let roi_rect = crate::models::Rect::default();
           let current_dpi = DpiInfo {
               display_id: "display1".to_string(),
               scale_factor: 1.5,
           };

           // Scale changed
           let stored_flags = crate::models::DpiInvalidationFlags {
               invalid: false,
               last_dpi_scale: 1.0,
               last_display_id: Some("display1".to_string()),
           };
           let result = check_dpi_invalidation(&roi_rect, &current_dpi, &stored_flags);
           assert!(result.invalid);
           assert_eq!(result.reason, Some("DPI 缩放已更改".to_string()));

           // Display changed
           let stored_flags = crate::models::DpiInvalidationFlags {
               invalid: false,
               last_dpi_scale: 1.5,
               last_display_id: Some("display2".to_string()),
           };
           let result = check_dpi_invalidation(&roi_rect, &current_dpi, &stored_flags);
           assert!(result.invalid);
           assert_eq!(result.reason, Some("显示器已更改".to_string()));

           // No change
           let stored_flags = crate::models::DpiInvalidationFlags {
               invalid: false,
               last_dpi_scale: 1.5,
               last_display_id: Some("display1".to_string()),
           };
           let result = check_dpi_invalidation(&roi_rect, &current_dpi, &stored_flags);
           assert!(!result.invalid);
       }
   }
   ```
